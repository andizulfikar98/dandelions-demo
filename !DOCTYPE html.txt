    <script>
        function toggleSla(id, action) {
            const badge = document.getElementById('badge1');
            const btnFreeze = document.getElementById('btn-freeze-1');
            const btnResume = document.getElementById('btn-resume-1');

            if (action === 'freeze') {
                badge.className = 'status-badge status-frozen';
                badge.innerText = '⏸ FROZEN';
                btnFreeze.style.display = 'none';
                btnResume.style.display = 'inline-block';
                alert('INSTRUKSI SISTEM: SLA untuk APP-2615900 telah di-Freeze (Argo Dijeda).');
            } else {
                badge.className = 'status-badge status-running';
                badge.innerText = '▶ RUNNING';
                btnFreeze.style.display = 'inline-block';
                btnResume.style.display = 'none';
                alert('INSTRUKSI SISTEM: Aplikasi di-Release. SLA kembali berjalan.');
            }
        }
